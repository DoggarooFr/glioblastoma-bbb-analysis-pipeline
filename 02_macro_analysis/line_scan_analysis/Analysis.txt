Sub PlotWithDynamicVerticalLineAndLabeling()
    Dim ws As Worksheet, dataWs As Worksheet
    Dim chartObj As ChartObject
    Dim xVal As Double
    Dim lineSeries As Series
    Dim i As Long
    Dim threshold As Double
    Dim maxRed As Double
    Dim maxIndex As Long
    Dim found As Boolean
    Dim redDecreaseIdx As Long
    Dim distances As Range
    Dim redValues As Range
    Dim greenValues As Range
    Dim minGreenLeft As Double
    Dim minGreenRight As Double
    Dim minGreen As Double
    Dim lastCol As Long
    Dim brownCol As Long
    Dim blueCol As Long
    Dim spacerCol As Long
    Dim dataSetCols As Collection
    Dim col As Variant
    Dim brownColExists As Boolean
    Dim blueColExists As Boolean
    Dim nonSalineValues As Collection
    Dim salineValues As Collection
    Dim dataArray As Variant

    ' Initialize collections to store values
    Set nonSalineValues = New Collection
    Set salineValues = New Collection

    ' Delete existing Data sheet if it exists
    On Error Resume Next
    Application.DisplayAlerts = False
    ThisWorkbook.Worksheets("Data").Delete
    Application.DisplayAlerts = True
    On Error GoTo 0

    ' Add a new Data sheet
    Set dataWs = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
    dataWs.Name = "Data"
    dataWs.Cells(1, 1).value = "Sheet Name"
    dataWs.Cells(1, 2).value = "Min Green Left (Brown)"
    dataWs.Cells(1, 3).value = "Min Green Right (Blue)"
    dataWs.Cells(1, 5).value = "Sheet Name (Saline)"
    dataWs.Cells(1, 6).value = "Min Green Left (Brown)"
    dataWs.Cells(1, 7).value = "Min Green Right (Blue)"

    ' Loop through each worksheet in the workbook
    For Each ws In ThisWorkbook.Worksheets
        ' Skip the Data sheet
        If ws.Name = "Data" Then GoTo SkipSheet

        ' Clear all existing charts
        For Each chartObj In ws.ChartObjects
            chartObj.Delete
        Next chartObj

        ' Determine the last column to add new columns for brown and blue values
        lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column

        ' Check if the columns with headers already exist
        brownColExists = False
        blueColExists = False
        For i = 1 To lastCol
            If ws.Cells(1, i).value = "Min Green Left (Brown)" Then
                brownCol = i
                brownColExists = True
            End If
            If ws.Cells(1, i).value = "Min Green Right (Blue)" Then
                blueCol = i
                blueColExists = True
            End If
        Next i

        ' If the columns don't exist, determine new column positions
        If Not brownColExists Then
            spacerCol = lastCol + 1
            brownCol = lastCol + 2
        End If
        If Not blueColExists Then
            If brownColExists Then
                blueCol = lastCol + 1
            Else
                blueCol = lastCol + 3
            End If
        End If

        ' Set headers for new columns if they don't already exist
        If Not brownColExists Then ws.Cells(1, brownCol).value = "Min Green Left (Brown)"
        If Not blueColExists Then ws.Cells(1, blueCol).value = "Min Green Right (Blue)"

        ' Clear the content of the columns except for the header
        If brownColExists Then ws.Range(ws.Cells(2, brownCol), ws.Cells(ws.Rows.Count, brownCol)).ClearContents
        If blueColExists Then ws.Range(ws.Cells(2, blueCol), ws.Cells(ws.Rows.Count, blueCol)).ClearContents

        ' Identify the columns containing data sets
        Set dataSetCols = New Collection
        For i = 1 To lastCol
            If ws.Cells(1, i).value Like "*Distance*" Then
                dataSetCols.Add i
            End If
        Next i

        For Each col In dataSetCols
            ' Define the ranges based on the column
            Set distances = ws.Range(ws.Cells(2, col), ws.Cells(ws.Cells(ws.Rows.Count, col).End(xlUp).Row, col))
            Set redValues = ws.Range(ws.Cells(2, col + 2), ws.Cells(ws.Cells(ws.Rows.Count, col).End(xlUp).Row, col + 2))
            Set greenValues = ws.Range(ws.Cells(2, col + 1), ws.Cells(ws.Cells(ws.Rows.Count, col).End(xlUp).Row, col + 1))

            ' Find the minimum green value that is not zero
            minGreen = FindNonZeroMin(greenValues)

            ' Check if the sheet name contains "saline"
            If InStr(ws.Name, "saline") > 0 Then
                ' Label the minimum Green value above the point
                Set chartObj = ws.ChartObjects.Add(Left:=100, Width:=750, Top:=50 + (350 * (Array_Index(dataSetCols, col) - 1)), Height:=300)
                With chartObj.Chart
                    ' Set chart type
                    .ChartType = xlXYScatterSmooth
                    .SetSourceData Source:=ws.Range(ws.Cells(1, col), ws.Cells(ws.Cells(ws.Rows.Count, col).End(xlUp).Row, col + 2))

                    ' Add Red data series
                    .SeriesCollection.NewSeries
                    With .SeriesCollection(1)
                        .Name = "Red"
                        .XValues = distances
                        .values = redValues
                        .Format.Line.ForeColor.RGB = RGB(255, 0, 0)
                        .MarkerStyle = xlMarkerStyleNone
                    End With

                    ' Add Green data series
                    .SeriesCollection.NewSeries
                    With .SeriesCollection(2)
                        .Name = "Green"
                        .XValues = distances
                        .values = greenValues
                        .Format.Line.ForeColor.RGB = RGB(0, 255, 0)
                        .MarkerStyle = xlMarkerStyleNone
                    End With

                    ' Label the minimum Green value above the point
                    With .SeriesCollection(2).Points(WorksheetFunction.Match(minGreen, greenValues, 0))
                        .ApplyDataLabels
                        With .DataLabel
                            .Position = xlLabelPositionAbove
                            .Font.Bold = True
                            .Font.Color = RGB(0, 0, 255) ' Blue color
                        End With
                    End With

                    ' Customize the chart
                    .HasTitle = True
                    .ChartTitle.Text = ws.Cells(2, col - 1).value
                    .HasLegend = False
                End With

                ' Add the min green value to the new column
                With ws.Cells(ws.Rows.Count, blueCol).End(xlUp).Offset(1)
                    .value = minGreen
                    .Font.Color = RGB(0, 0, 255) ' Blue color
                End With

                ' Add NaN to the brown column if no left min green value
                With ws.Cells(ws.Rows.Count, brownCol).End(xlUp).Offset(1)
                    .value = "NaN"
                    .Font.Color = RGB(165, 42, 42) ' Brown color
                End With

                ' Store the minimum values for saline sheets
                salineValues.Add Array(ws.Name, minGreen, "NaN")

            Else
                ' Find the maximum value in the Red column
                maxRed = WorksheetFunction.Max(redValues)
                maxIndex = WorksheetFunction.Match(maxRed, redValues, 0)
                threshold = maxRed * 0.2
                found = False

                ' Find the x-value where Red decreases by 95% from the maximum
                For i = maxIndex To redValues.Rows.Count
                    If redValues.Cells(i, 1).value <= threshold Then
                        xVal = distances.Cells(i, 1).value
                        redDecreaseIdx = i
                        found = True
                        Exit For
                    End If
                Next i

                ' Create the chart
                Set chartObj = ws.ChartObjects.Add(Left:=100, Width:=750, Top:=50 + (350 * (Array_Index(dataSetCols, col) - 1)), Height:=300)
                With chartObj.Chart
                    ' Set chart type
                    .ChartType = xlXYScatterSmooth
                    .SetSourceData Source:=ws.Range(ws.Cells(1, col), ws.Cells(ws.Cells(ws.Rows.Count, col).End(xlUp).Row, col + 2))

                    ' Add Red data series
                    .SeriesCollection.NewSeries
                    With .SeriesCollection(1)
                        .Name = "Red"
                        .XValues = distances
                        .values = redValues
                        .Format.Line.ForeColor.RGB = RGB(255, 0, 0)
                        .MarkerStyle = xlMarkerStyleNone
                    End With

                    ' Add Green data series
                    .SeriesCollection.NewSeries
                    With .SeriesCollection(2)
                        .Name = "Green"
                        .XValues = distances
                        .values = greenValues
                        .Format.Line.ForeColor.RGB = RGB(0, 255, 0)
                        .MarkerStyle = xlMarkerStyleNone
                    End With

                    ' Add a series for the vertical line if found
                    If found Then
                        .SeriesCollection.NewSeries
                        Set lineSeries = .SeriesCollection(3)
                        With lineSeries
                            .Name = "Vertical Line"
                            .XValues = Array(xVal, xVal)
                            .values = Array(0, WorksheetFunction.Max(ws.Range(redValues.Address, greenValues.Address)))
                            .Border.LineStyle = xlDash
                            .Format.Line.ForeColor.RGB = RGB(0, 0, 0)
                            .MarkerStyle = xlMarkerStyleNone
                        End With

                        ' Find the minimum Green value on the left and right of the xVal
                        minGreenLeft = FindNonZeroMin(greenValues.Resize(redDecreaseIdx))
                        If greenValues.Rows.Count > redDecreaseIdx Then
                            minGreenRight = FindNonZeroMin(greenValues.Offset(redDecreaseIdx).Resize(greenValues.Rows.Count - redDecreaseIdx))
                        Else
                            minGreenRight = minGreenLeft ' Avoid error by setting it to the left min
                        End If

                        ' Label the minimum Green values above the points
                        With .SeriesCollection(2).Points(WorksheetFunction.Match(minGreenLeft, greenValues.Resize(redDecreaseIdx), 0))
                            .ApplyDataLabels
                            With .DataLabel
                                .Position = xlLabelPositionAbove
                                .Font.Bold = True
                                .Font.Color = RGB(165, 42, 42) ' Brown color
                            End With
                        End With

                        ' Add the left min green value to the new column
                        With ws.Cells(ws.Rows.Count, brownCol).End(xlUp).Offset(1)
                            .value = minGreenLeft
                            .Font.Color = RGB(165, 42, 42) ' Brown color
                        End With

                        ' Corrected: Offset and Resize properly
                        If greenValues.Rows.Count > redDecreaseIdx Then
                            With .SeriesCollection(2).Points(WorksheetFunction.Match(minGreenRight, greenValues.Offset(redDecreaseIdx).Resize(greenValues.Rows.Count - redDecreaseIdx), 0) + redDecreaseIdx)
                                .ApplyDataLabels
                                With .DataLabel
                                    .Position = xlLabelPositionAbove
                                    .Font.Bold = True
                                    .Font.Color = RGB(0, 0, 255) ' Blue color
                                End With
                            End With
                            ' Add the right min green value to the new column
                            With ws.Cells(ws.Rows.Count, blueCol).End(xlUp).Offset(1)
                                .value = minGreenRight
                                .Font.Color = RGB(0, 0, 255) ' Blue color
                            End With
                        Else
                            ' Add NaN to the blue column if no right min green value
                            With ws.Cells(ws.Rows.Count, blueCol).End(xlUp).Offset(1)
                                .value = "NaN"
                                .Font.Color = RGB(0, 0, 255) ' Blue color
                            End With
                        End If
                    Else
                        ' Find the minimum Green value
                        minGreen = FindNonZeroMin(greenValues)

                        ' Label the minimum Green value above the point
                        With .SeriesCollection(2).Points(WorksheetFunction.Match(minGreen, greenValues, 0))
                            .ApplyDataLabels
                            With .DataLabel
                                .Position = xlLabelPositionAbove
                                .Font.Bold = True
                                .Font.Color = RGB(0, 0, 255) ' Blue color
                            End With
                        End With

                        ' Add the min green value to the new column
                        With ws.Cells(ws.Rows.Count, blueCol).End(xlUp).Offset(1)
                            .value = minGreen
                            .Font.Color = RGB(0, 0, 255) ' Blue color
                        End With

                        ' Add NaN to the brown column if no left min green value
                        With ws.Cells(ws.Rows.Count, brownCol).End(xlUp).Offset(1)
                            .value = "NaN"
                            .Font.Color = RGB(165, 42, 42) ' Brown color
                        End With
                    End If

                    ' Store the minimum values for non-saline sheets
                    nonSalineValues.Add Array(ws.Name, minGreenLeft, minGreenRight)
                
                ' Customize the chart
                .HasTitle = True
                .ChartTitle.Text = ws.Cells(2, col - 1).value
                .HasLegend = False
            End With
        End If
    Next col

SkipSheet:
    Next ws

    ' Compile the stored values into the Data sheet
    Dim value As Variant
    Dim rowIndex As Long
    
    rowIndex = 2
    For Each value In nonSalineValues
        dataWs.Cells(rowIndex, 1).value = value(0)
        dataWs.Cells(rowIndex, 2).value = value(1)
        dataWs.Cells(rowIndex, 3).value = value(2)
        rowIndex = rowIndex + 1
    Next value

    rowIndex = 2
    For Each value In salineValues
        dataWs.Cells(rowIndex, 5).value = value(0)
        dataWs.Cells(rowIndex, 6).value = value(1)
        dataWs.Cells(rowIndex, 7).value = value(2)
        rowIndex = rowIndex + 1
    Next value
End Sub

Function Array_Index(arr As Collection, value As Variant) As Long
    Dim i As Long
    For i = 1 To arr.Count
        If arr(i) = value Then
            Array_Index = i
            Exit Function
        End If
    Next i
    Array_Index = -1 ' Not found
End Function

Function FindNonZeroMin(rng As Range) As Double
    Dim minVal As Double
    Dim cell As Range
    Dim values() As Double
    Dim j As Long

    ' Collect all non-zero values
    j = 0
    For Each cell In rng
        If cell.value <> 0 Then
            ReDim Preserve values(j)
            values(j) = cell.value
            j = j + 1
        End If
    Next cell

    ' If there are non-zero values, find the minimum
    If j > 0 Then
        minVal = Application.WorksheetFunction.Min(values)
        FindNonZeroMin = minVal
    Else
        FindNonZeroMin = 0 ' Handle case where all values are zero
    End If
End Function
