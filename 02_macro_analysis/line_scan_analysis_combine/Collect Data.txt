
Sub CompileAllMinGreenValues()
    Dim dataWs As Worksheet
    Dim ws As Worksheet
    Dim lastCol As Long
    Dim lastRow As Long
    Dim brownCol As Long
    Dim blueCol As Long
    Dim brownColExists As Boolean
    Dim blueColExists As Boolean
    Dim dataRow As Long
    Dim i As Long
    Dim j As Long
    Dim dataLength As Long
    
    ' Add a new Data sheet
    On Error Resume Next
    Set dataWs = ThisWorkbook.Worksheets("Data")
    On Error GoTo 0
    If Not dataWs Is Nothing Then
        Application.DisplayAlerts = False
        dataWs.Delete
        Application.DisplayAlerts = True
    End If
    
    Set dataWs = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
    dataWs.Name = "Data"
    dataWs.Cells(1, 1).Value = "Sheet Name"
    dataWs.Cells(1, 2).Value = "Min Green Left (Brown)"
    dataWs.Cells(1, 3).Value = "Min Green Right (Blue)"
    
    dataRow = 2
    
    ' Loop through each worksheet in the workbook
    For Each ws In ThisWorkbook.Worksheets
        ' Skip the Data sheet and Saline sheets
        If ws.Name = "Data" Or InStr(ws.Name, "Saline") > 0 Then GoTo SkipSheet
        
        ' Find the last column and row in the current worksheet
        lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column
        lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
        
        brownColExists = False
        blueColExists = False
        
        ' Check for the existence of the required columns
        For i = 1 To lastCol
            If ws.Cells(1, i).Value = "Min Green Left (Brown)" Then
                brownCol = i
                brownColExists = True
            End If
            If ws.Cells(1, i).Value = "Min Green Right (Blue)" Then
                blueCol = i
                blueColExists = True
            End If
        Next i
        
        ' If both columns exist, determine the length of data
        If brownColExists And blueColExists Then
            dataLength = ws.Cells(ws.Rows.Count, brownCol).End(xlUp).Row - 1 ' Excluding header row
            
            ' Append the sheet name and data
            For j = 2 To 1 + dataLength ' Assuming the data starts from row 2
                dataWs.Cells(dataRow, 1).Value = ws.Name
                dataWs.Cells(dataRow, 2).Value = ws.Cells(j, brownCol).Value
                dataWs.Cells(dataRow, 3).Value = ws.Cells(j, blueCol).Value
                dataRow = dataRow + 1
            Next j
        End If
        
SkipSheet:
    Next ws

End Sub
